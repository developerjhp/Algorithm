# A)
```js
function solution(left, right) {
    let res = 0;
    for (let i = left; i <= right; i++) {
        if (getNumber(i)%2) {
            res = res - i;
        }
        else {
            res = res + i;
        }
    }
    return res;
}

const getNumber = (el) => {
    let res = 0;
    for(let i = 1; i <= el; i++) {
        if (el%i === 0) {
            res++;
        }
    }
    return res;
}
```
